# הוראות מלאות לAI — תיקון כל הבעיות בפרויקט

## הקשר כללי
זהו monorepo של פלטפורמת AI App Builder עם המבנה הבא:
- `artifacts/api-server` — Express 5 + WebSocket (port 8080)
- `artifacts/app-builder` — React + Vite (port 20311)
- `lib/db` — Drizzle ORM + PostgreSQL schemas
- `lib/api-zod` — Zod validators
- `lib/integrations-anthropic-ai` — Anthropic SDK wrapper

עבור על כל הבעיות הבאות לפי הסדר ותקן אותן. עבור כל תיקון — כתוב את הקוד המלא של הקובץ הרלוונטי.

### בעיה 11: Authorization לא נאכף ב-routes

```typescript
export async function requireProjectAccess(req, res, next) {
  if (!req.isAuthenticated()) return res.status(401).json({ error: 'Unauthorized' });
  
  const projectId = Number(req.params.id);
  const project = await db.select().from(projectsTable)
    .where(eq(projectsTable.id, projectId)).limit(1);
  
  if (!project[0]) return res.status(404).json({ error: 'Not found' });
  
  const isOwner = project[0].userId === req.user.id;
  const teamMember = await db.select().from(teamsTable)
    .where(and(eq(teamsTable.projectId, projectId), eq(teamsTable.userId, req.user.id)))
    .limit(1);
  
  if (!isOwner && !teamMember[0]) return res.status(403).json({ error: 'Forbidden' });
  
  req.project = project[0];
  req.teamRole = isOwner ? 'owner' : teamMember[0].role;
  next();
}

export function requireRole(...roles: string[]) {
  return (req, res, next) => {
    if (!roles.includes(req.teamRole)) return res.status(403).json({ error: 'Insufficient permissions' });
    next();
  };
}
```
החל את ה-middleware על כל project routes.

```typescript
// artifacts/api-server/src/lib/cleanup.ts
import { db, sessionsTable, projectMessagesTable, analyticsEventsTable } from '@workspace/db';
import { lt, sql } from 'drizzle-orm';

export async function runCleanup() {
  const now = new Date();
  
  // מחק sessions שפגו
  await db.delete(sessionsTable).where(lt(sessionsTable.expire, now));
  
  // מחק messages ישנים מעל 500 הודעות לפרויקט (שמור רק 500 אחרונות)
  await db.execute(sql`
    DELETE FROM project_messages WHERE id NOT IN (
      SELECT id FROM project_messages pm2
      WHERE pm2.project_id = project_messages.project_id
      ORDER BY created_at DESC LIMIT 500
    )
  `);
  
  // מחק analytics events ישנים מעל 90 יום
  const ninetyDaysAgo = new Date(Date.now() - 90 * 24 * 60 * 60 * 1000);
  await db.delete(analyticsEventsTable).where(lt(analyticsEventsTable.createdAt, ninetyDaysAgo));
  
  console.log('[Cleanup] Done');
}

// הפעל פעם ביום

```

```typescript
async function extractAndSaveDNAWithRetry(projectId: number, response: string, mode: string) {
  for (let attempt = 0; attempt < 3; attempt++) {
    try {
      await extractAndSaveDNA(projectId, response, mode);
      return;
```

### בעיה 19: אין semantic caching
**קובץ חדש:** `artifacts/api-server/src/lib/prompt-cache.ts`

**מה לעשות:**
```typescript
import { db } from '@workspace/db';
import crypto from 'crypto';

// Cache פשוט בDB — לא Redis, לא מורכב
export async function getCachedResponse(prompt: string, mode: string): Promise<string | null> {
  const hash = crypto.createHash('sha256')
    .update(prompt.slice(0, 500) + mode) // hash על 500 תווים ראשונים + mode
    .digest('hex');
  
  const cached = await db.execute(
    sql`SELECT response FROM prompt_cache WHERE hash = ${hash} AND created_at > NOW() - INTERVAL '24 hours' LIMIT 1`
  );
  return cached.rows[0]?.response ?? null;
}

export async function setCachedResponse(prompt: string, mode: string, response: string) {
  const hash = crypto.createHash('sha256').update(prompt.slice(0, 500) + mode).digest('hex');
  await db.execute(
    sql`INSERT INTO prompt_cache (hash, response) VALUES (${hash}, ${response}) ON CONFLICT (hash) DO UPDATE SET response = ${response}, created_at = NOW()`
  );
}
```

### בעיה 22: אין fallback ל-manifest parsing

```typescript
function parseManifestOrFallback(response: string, stack: string): ParsedFiles | null {
  // נסה parse רגיל
  const parsed = parseFileManifest(response);
  if (parsed && parsed.length > 0) return parsed;
  
  // fallback: נסה לחלץ code blocks
  const codeBlocks = response.match(/```(?:tsx?|jsx?|html|css)\n([\s\S]*?)```/g);
  if (codeBlocks && codeBlocks.length > 0) {
    console.warn('[Manifest] Fallback to code block extraction');
    return codeBlocks.map((block, i) => ({
      path: `src/component_${i}.tsx`,
      content: block.replace(/```\w*\n/, '').replace(/```$/, ''),
      language: 'typescript',
      isEntrypoint: i === 0
    }));
  }
  
  // last resort: treat entire response as HTML
  if (stack === 'html') {
    return [{ path: 'index.html', content: response, language: 'html', isEntrypoint: true }];
  }
  
  return null;
}
```

```typescript
});

// Retry wrapper
export async function withRetry<T>(fn: () => Promise<T>, retries = 3): Promise<T> {
  for (let i = 0; i < retries; i++) {
    try {
      return await fn();
    } catch (err: any) {
      if (i === retries - 1) throw err;
      if (err.code === 'ECONNREFUSED' || err.code === '57P01') {
        await new Promise(r => setTimeout(r, 1000 * (i + 1)));
        continue;
      }
      throw err;
    }
  }
  throw new Error('Unreachable');
}
```

```typescript
  state: State = { hasError: false };
  
  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error };
  }

    console.error('[ErrorBoundary]', error, info);
  }
  
  render() {

      return this.props.fallback ?? (

          <div className="text-center">
            <p className="font-semibold">משהו השתבש</p>

              className="mt-2 px-3 py-1 bg-red-500/20 rounded text-xs">
              נסה שוב
            </button>
          </div>
        </div>
      );
```

```typescript
  const sub = await getUserSubscription(req.user.id);
  const plan = PLANS[sub.planId];
  
  // בדוק project limit
  if (req.method === 'POST' && req.path === '/api/v1/projects') {
    const count = await getProjectCount(req.user.id);
    if (count >= plan.maxProjects) {
      return res.status(403).json({ 
        error: 'הגעת למגבלת הפרויקטים בתוכנית שלך',
        upgradeUrl: '/pricing'
      });
    }
  }
  
  next();
}
```
