## תיקון 1 — Terminal: הסר את חלון הפרודקשן האוטומטי

**קובץ:** `artifacts/api-server/src/routes/terminal.ts`

**הבעיה:** שורות 10–12:
```ts
const TERMINAL_ENABLED =
  process.env["ENABLE_TERMINAL"] === "true" ||
  process.env["NODE_ENV"] !== "production";   // ← הטרמינל דולק אוטומטית ב-dev/staging
```

**תיקון — החלף את ההגדרה ל:**
```ts
const TERMINAL_ENABLED = process.env["ENABLE_TERMINAL"] === "true";
```

**הסבר:** הטרמינל פותח bash shell אמיתי ללא container isolation בין משתמשים. בכל סביבה שאינה Replit (staging, dev local, preview deploy) הוא דולק אוטומטית. הפעלה מפורשת בלבד (`ENABLE_TERMINAL=true`) היא דרישת מינימום עד שיש sandbox אמיתי.

הוסף גם הערה מעל ההגדרה:
```ts
// SECURITY: Terminal opens a real bash shell. No per-user container isolation exists.
// Only enable in Replit (managed sandboxing) or when running inside Docker per-user containers.
// Never enable in shared-server deployments.
```

## תיקון 2 — Cookie sameSite: "lax" → "strict"

**קובץ:** `artifacts/api-server/src/routes/auth.ts`

**הבעיה:** שורות 27–33 ו-37–43. שתי הפונקציות `setSessionCookie` ו-`setOidcCookie` משתמשות ב-`sameSite: "lax"`.

**תיקון — שנה את שתי הפונקציות:**
```ts
function setSessionCookie(res: Response, sid: string) {
  res.cookie(SESSION_COOKIE, sid, {
    httpOnly: true,
    secure: true,
    sameSite: "strict",   // ← שינוי מ-"lax"
    path: "/",
    maxAge: SESSION_TTL,
  });
}

function setOidcCookie(res: Response, name: string, value: string) {
  res.cookie(name, value, {
    httpOnly: true,
    secure: true,
    sameSite: "strict",   // ← שינוי מ-"lax"
    path: "/",
    maxAge: OIDC_COOKIE_TTL,
  });
}
```

**הסבר:** `sameSite: "lax"` מאפשר שליחת cookies בניווט GET ממקורות חיצוניים (לינקים, redirects). `strict` חוסם לחלוטין cross-site. כיוון שכל ה-auth הוא OIDC עם redirect מ-Replit, `strict` עובד תקין ומגן מ-CSRF.

## תיקון 9 — פרק את ai-generation.ts: חלץ snapshot writer

**הבעיה:** `ai-generation.ts` הוא 1,023 שורות. שלב אחד ראשון — חלץ את לוגיקת ה-DB write לקובץ נפרד.

**קובץ חדש:** `artifacts/api-server/src/routes/projects/messages/snapshot-writer.ts`

```ts
import { eq } from "drizzle-orm";
import {
  db,
  projectsTable,
  projectMessagesTable,
  projectSnapshotsTable,
  projectFilesTable,
} from "@workspace/db";
import type { ChangeSummary } from "../../../services/ai/change-summary";

export interface SnapshotWriteParams {
  projectId: number;
  aiResponse: string;
  previewHtml: string | null;
  stack: string | null;
  files?: { path: string; content: string }[];
  changeSummary: ChangeSummary | null;
  isFirstMessage: boolean;
}

export interface SnapshotWriteResult {
  snapshotId: number;
  messageId: number;
}

/**
 * Atomically saves:
 * 1. AI assistant message to project_messages
 * 2. Project snapshot to project_snapshots
 * 3. Project files (for React projects)
 * 4. Updated previewHtml on the project
 */
export async function writeSnapshot(
  params: SnapshotWriteParams,
): Promise<SnapshotWriteResult> {
  const {
    projectId,
    aiResponse,
    previewHtml,
    stack,
    files,
    changeSummary,
    isFirstMessage,
  } = params;

  return await db.transaction(async (tx) => {
    // 1. Save assistant message
    const [msg] = await tx
      .insert(projectMessagesTable)
      .values({ projectId, role: "assistant", content: aiResponse })
      .returning({ id: projectMessagesTable.id });

    // 2. Save snapshot
    const [snap] = await tx
      .insert(projectSnapshotsTable)
      .values({
        projectId,
        html: previewHtml ?? "",
        diffStats: changeSummary as unknown as Record<string, unknown> | null,
      })
      .returning({ id: projectSnapshotsTable.id });

    // 3. Update project files (React/multi-file)
    if (files && files.length > 0) {
      await tx
        .delete(projectFilesTable)
        .where(eq(projectFilesTable.projectId, projectId));
      await tx.insert(projectFilesTable).values(
        files.map((f) => ({ projectId, path: f.path, content: f.content })),
      );
    }

    // 4. Update project record
    const updateData: Record<string, unknown> = {
      updatedAt: new Date(),
      ...(previewHtml !== null && { previewHtml }),
      ...(isFirstMessage && { firstGeneratedAt: new Date() }),
    };
    await tx
      .update(projectsTable)
      .set(updateData)
      .where(eq(projectsTable.id, projectId));

    return { snapshotId: snap.id, messageId: msg.id };
  });
}
```

**לאחר יצירת הקובץ**, בתוך `ai-generation.ts` בשורה ~742 (ה-atomic DB transaction), החלף את כל בלוק ה-transaction בקריאה:
```ts

  projectId: params.id,
  aiResponse: fullResponse,
  previewHtml: extractedHtml,
  stack: project.stack ?? null,
  files: reactFiles,

  isFirstMessage,
});

## תיקון 10 — Image URL validation לפני הזרקה לpreview

**קובץ:** `artifacts/api-server/src/services/ai/preview.ts`

**הוסף את הפונקציה לפני הייצוא הראשי:**
```ts
/**
 * Strips img src attributes that point to suspicious or potentially harmful URLs.
 * Allows: https:// URLs and relative paths. Blocks: javascript:, data:, and unknown protocols.
 */
export function sanitizeImageUrls(html: string): string {
  return html.replace(
    /<img([^>]*)\ssrc=["']([^"']+)["']([^>]*)>/gi,
    (match, before, src, after) => {
      const lower = src.toLowerCase().trim();
      // Allow https and relative paths
      if (
        lower.startsWith("https://") ||
        lower.startsWith("/") ||
        lower.startsWith("./") ||
        lower.startsWith("../")
      ) {
        return match;
      }
      // Block javascript:, data:, http: (non-https), and anything else
      return `<img${before} src=""${after}>`;
    },
  );
}
```

**בקובץ `ai-generation.ts`**, אחרי שליפת ה-HTML מClaude ולפני השמירה, הוסף:

}
```

export function initSentry(): void {

    },
  });
}

## תיקון 12 — secret rotation: הוסף תיעוד ו-utility script

**קובץ חדש:** `docs/security/secret-rotation.md`

```markdown
# Secret Rotation Guide

## ENCRYPTION_KEY rotation

The ENCRYPTION_KEY encrypts all stored GitHub tokens, Netlify tokens, and Vercel tokens.

### Steps to rotate:

1. Generate a new key:
   ```bash
   node -e "console.log(require('crypto').randomBytes(32).toString('hex'))"
   ```

2. Run the migration script BEFORE changing the env var:
   ```bash
   OLD_KEY=<current_key> NEW_KEY=<new_key> pnpm --filter @workspace/scripts run rotate-secrets
   ```

3. Once migration completes successfully, update ENCRYPTION_KEY in your environment.

4. Verify: run a test deploy to confirm tokens still work.

### Emergency (key compromised):
1. Immediately revoke all GitHub PATs, Netlify tokens, Vercel tokens in those platforms.
2. Ask all users to reconnect their integrations.
3. Set a new ENCRYPTION_KEY.

## SESSION_SECRET rotation
Sessions are stored in DB (not JWT). Rotating SESSION_SECRET invalidates all active sessions —
all users will be logged out. Safe to rotate at any time if needed.
```

**קובץ חדש:** `scripts/src/rotate-secrets.ts`

```ts
import { db, projectSecretsTable } from "@workspace/db";
import { createCipheriv, createDecipheriv, randomBytes } from "node:crypto";

const OLD_KEY = process.env["OLD_KEY"];
const NEW_KEY = process.env["NEW_KEY"];

if (!OLD_KEY || !NEW_KEY) {
  console.error("Usage: OLD_KEY=<hex> NEW_KEY=<hex> pnpm run rotate-secrets");
  process.exit(1);
}

function decrypt(encryptedHex: string, key: string): string {
  const buf = Buffer.from(encryptedHex, "hex");
  const iv = buf.subarray(0, 16);
  const tag = buf.subarray(16, 32);
  const ciphertext = buf.subarray(32);
  const decipher = createDecipheriv("aes-256-gcm", Buffer.from(key, "hex"), iv);
  decipher.setAuthTag(tag);
  return decipher.update(ciphertext, undefined, "utf8") + decipher.final("utf8");
}

function encrypt(plaintext: string, key: string): string {
  const iv = randomBytes(16);
  const cipher = createCipheriv("aes-256-gcm", Buffer.from(key, "hex"), iv);
  const encrypted = Buffer.concat([cipher.update(plaintext, "utf8"), cipher.final()]);
  const tag = cipher.getAuthTag();
  return Buffer.concat([iv, tag, encrypted]).toString("hex");
}

async function main() {
  const secrets = await db.select().from(projectSecretsTable);
  console.log(`Rotating ${secrets.length} secrets...`);

  let success = 0, failed = 0;
  for (const secret of secrets) {
    try {
      const plaintext = decrypt(secret.encryptedValue, OLD_KEY!);
      const reencrypted = encrypt(plaintext, NEW_KEY!);
      await db
        .update(projectSecretsTable)
        .set({ encryptedValue: reencrypted })
        .where(/* eq(projectSecretsTable.id, secret.id) */ undefined as any);
      success++;
    } catch {
      console.error(`Failed to rotate secret id=${secret.id}`);
      failed++;
    }
  }
  console.log(`Done: ${success} rotated, ${failed} failed`);
  if (failed > 0) process.exit(1);
}

main().catch(console.error);
```
