## תיקון 1 — Parallel Agent Pipeline

**קובץ:** `artifacts/api-server/src/agents/agent_orchestrator.ts`
**בעיה:** Frontend ו-Backend רצים ברצף למרות שאינם תלויים זה בזה. DevOps ו-QA גם הם יכולים לרוץ במקביל.
**השפעה:** חיסכון של ~40% בזמן המתנה למשתמש.

**מצא את הקוד הזה (שורות Phase 2–5):**

```typescript
// ── Phase 2: Frontend ──────────────────────────────────────

agentOutputs.push(frontendOutput);

// ── Phase 3: Backend (sees architect + frontend) ───────────

agentOutputs.push(backendOutput);

// ── Phase 4: DevOps (sees all previous) ───────────────────
const devopsOutput = await runSpecialistAgent("devops", ...)
agentOutputs.push(devopsOutput);

// ── Phase 5: QA (sees everything) ─────────────────────────
const qaOutput = await runSpecialistAgent("qa", ...)
agentOutputs.push(qaOutput);
```

**החלף ב:**

```typescript
// ── Phase 2+3: Frontend & Backend במקביל ──────────────────

    "תשתיות ופריסה",
```

### שלב א׳ — צור קובץ חדש: `services/messages/plan-guard.ts`

העבר לשם את כל הקוד של בדיקת plan limits (כ-30 שורות שמתחילות ב-`const PLAN_LIMITS`):

```typescript
// artifacts/api-server/src/services/messages/plan-guard.ts
import {
  db,
  userSubscriptionsTable,
  projectMessagesTable,
} from "@workspace/db";
import { eq, and } from "drizzle-orm";
import type { Response } from "express";

interface PlanGuardResult {
  allowed: boolean;
  plan: string;
}

const PLAN_LIMITS: Record<string, { maxMessages: number; name: string }> = {
  free: { maxMessages: 50, name: "Free" },
  pro: { maxMessages: 200, name: "Pro" },
  studio: { maxMessages: 2000, name: "Studio" },
};

export async function enforcePlanLimit(
  userId: string,
  projectId: string,
  existingUserMessageCount: number,
  res: Response,
): Promise<PlanGuardResult> {
  const planId = await db
    .select({ planId: userSubscriptionsTable.planId })
    .from(userSubscriptionsTable)
    .where(eq(userSubscriptionsTable.userId, userId))
    .then((rows) => rows[0]?.planId ?? "free")
    .catch(() => "free");

  const limits = PLAN_LIMITS[planId] ?? PLAN_LIMITS.free!;

  if (existingUserMessageCount >= limits.maxMessages) {
    res.status(429).json({
      error: `הגעת למגבלת ${limits.maxMessages} הודעות של פלן ${limits.name}. שדרג את החשבון כדי להמשיך.`,
      limitReached: true,
      plan: planId,
      limit: limits.maxMessages,
    });
    return { allowed: false, plan: planId };
  }
  return { allowed: true, plan: planId };
}
```

### שלב ב׳ — צור קובץ חדש: `services/messages/post-generation.ts`

העבר לשם את כל לוגיקת ה-"אחרי יצירת הקוד" (QA auto-trigger, DNA extraction, next-step, suggestions):

```typescript
// artifacts/api-server/src/services/messages/post-generation.ts
import { setImmediate } from "timers";
import { anthropic } from "@workspace/integrations-anthropic-ai";
import { db, qaTestResultsTable, projectDnaTable } from "@workspace/db";
import { eq } from "drizzle-orm";
import {
  extractAndSaveDNA,
  extractAndSaveMemoryChunks,
  shouldExtractMemory,
} from "../memory/project-dna";
import { detectContextualSuggestions, detectNextStep } from "../ai/intent";
import type { SendEvent } from "./types"; // טיפוס שתגדיר לפונקציית sendEvent

export interface PostGenerationInput {
  projectId: string;
  extractedHtml: string | null;
  fullResponse: string;
  userMessage: string;
  existingMessagesCount: number;
  integrations?: Record<string, string>;
  sendEvent: SendEvent;
}

export function runPostGeneration(input: PostGenerationInput): void {
  const {
    projectId,
    extractedHtml,
    fullResponse,
    userMessage,
    existingMessagesCount,
    integrations,
    sendEvent,
  } = input;

  const hasUpdate = !!extractedHtml;

  // Contextual suggestions
  const codeForSuggestions = extractedHtml ?? fullResponse;
  if (codeForSuggestions) {
    const suggestions = detectContextualSuggestions(
      codeForSuggestions,
      integrations,
    );
    if (suggestions.length > 0) sendEvent({ type: "suggestions", suggestions });
  }

  // Next-step engine
  if (hasUpdate && codeForSuggestions) {
    const lang = /[\u0590-\u05FF]/.test(userMessage) ? "he" : "en";
    const nextStep = detectNextStep(
      userMessage,
      codeForSuggestions,
      existingMessagesCount,
      lang,
    );
    if (nextStep) sendEvent({ type: "next_step", text: nextStep });
  }

  // QA auto-trigger + DNA extraction — fire-and-forget
  if (hasUpdate && extractedHtml) {
    setImmediate(async () => {
      try {
        // QA
        const code = extractedHtml.slice(0, 8000);
        const msg = await anthropic.messages.create({
          model: "claude-haiku-4-5-20251001",
          max_tokens: 800,
          system:
            "You are a QA engineer. Return only valid JSON, no explanation.",
          messages: [
            {
              role: "user",
              content: `Analyze this HTML and return JSON: {"quality_score": number, "critical_issues": string[], "auto_fix_suggestions": [{"issue":string,"fix":string}]}.\n\nCode:\n${code}`,
            },
          ],
        });
        const text =
          msg.content[0].type === "text" ? msg.content[0].text : "{}";
        const qa = JSON.parse(text.match(/\{[\s\S]*\}/)?.[0] ?? "{}") as {
          quality_score?: number;
          auto_fix_suggestions?: Array<{ issue: string; fix: string }>;
        };
        if (qa.quality_score !== undefined) {
          const score = Number(qa.quality_score) || 0;
          await db
            .insert(qaTestResultsTable)
            .values({
              projectId,
              testSuite: { auto_triggered: true, ...qa },
              coveragePercent: score,
              autoFixSuggestions: qa.auto_fix_suggestions ?? [],
              passed: score >= 70 ? 1 : 0,
              failed: score < 70 ? 1 : 0,
            })
            .catch(() => {});
        }

        // DNA extraction
        if (shouldExtractMemory(existingMessagesCount)) {
          await extractAndSaveDNA(projectId, extractedHtml).catch(() => {});
          await extractAndSaveMemoryChunks(
            projectId,
            userMessage,
            extractedHtml,
          ).catch(() => {});
        }
      } catch {
        /* non-critical */
      }
    });
  }
}
```

## תיקון 3 — תיקון אבטחה בהצפנה (Static Salt)

**קובץ:** `artifacts/api-server/src/lib/encryption.ts`
**בעיה:** ה-salt הוא קבוע `"builder-ai-secrets-salt-v1"` לכל ה-keys. Salt צריך להיות אקראי לכל הצפנה, ולהישמר יחד עם ה-ciphertext.

**מחק את השורה:**

```typescript
const SALT = "builder-ai-secrets-salt-v1";
```

**מחק את הפונקציה `getKey()` כולה.**

**החלף את `encrypt` ו-`decrypt` בגרסאות הבאות:**

```typescript
const ALGORITHM = "aes-256-gcm";
const IV_LENGTH = 16;
const SALT_LENGTH = 32;
const TAG_LENGTH = 16;
const KEY_LENGTH = 32;

function deriveKey(secret: string, salt: Buffer): Buffer {
  return scryptSync(secret, salt, KEY_LENGTH);
}

function getMasterSecret(): string {
  const secret = process.env.ENCRYPTION_KEY;
  if (!secret) {
    if (process.env.NODE_ENV === "production") {
      throw new Error("[Security] ENCRYPTION_KEY is required in production.");
    }
    console.warn("[Security] ENCRYPTION_KEY not set — dev-only insecure key.");
    return "dev-only-insecure-key-never-use-in-prod";
  }
  if (secret.length < 32) {
    throw new Error(
      "[Security] ENCRYPTION_KEY must be at least 32 characters.",
    );
  }
  return secret;
}

export function encrypt(plaintext: string): string {
  const secret = getMasterSecret();
  const salt = randomBytes(SALT_LENGTH); // ← salt אקראי לכל הצפנה
  const key = deriveKey(secret, salt);
  const iv = randomBytes(IV_LENGTH);
  const cipher = createCipheriv(ALGORITHM, key, iv);
  const encrypted = Buffer.concat([
    cipher.update(plaintext, "utf8"),
    cipher.final(),
  ]);
  const tag = cipher.getAuthTag();
  // פורמט: salt(32) + iv(16) + tag(16) + ciphertext
  return Buffer.concat([salt, iv, tag, encrypted]).toString("base64url");
}

export function decrypt(ciphertext: string): string {
  const secret = getMasterSecret();
  const buf = Buffer.from(ciphertext, "base64url");
  const salt = buf.subarray(0, SALT_LENGTH);
  const iv = buf.subarray(SALT_LENGTH, SALT_LENGTH + IV_LENGTH);
  const tag = buf.subarray(
    SALT_LENGTH + IV_LENGTH,
    SALT_LENGTH + IV_LENGTH + TAG_LENGTH,
  );
  const encrypted = buf.subarray(SALT_LENGTH + IV_LENGTH + TAG_LENGTH);
  const key = deriveKey(secret, salt);
  const decipher = createDecipheriv(ALGORITHM, key, iv);
  decipher.setAuthTag(tag);
  return decipher.update(encrypted).toString("utf8") + decipher.final("utf8");
}
```

> **⚠️ שים לב:** תיקון זה שובר תאימות לאחור — כל הסודות המוצפנים הקיימים בבסיס הנתונים יצטרכו להיות מוצפנים מחדש. הוסף migration script שמפענח עם הגרסה הישנה ומצפין מחדש עם החדשה.

## תיקון 4 — LLM-based Intent Detection לעברית

**קובץ:** `artifacts/api-server/src/agents/definitions.ts`
**בעיה:** `classifyIntent()` משתמש ב-regex. עברית לא מתאימה לרוב הביטויים — "תבנה לי אתר" לא יתאים לאף pattern ויחזיר ברירת מחדל `"frontend"`.

**מצא את הפונקציה `classifyIntent` והחלף אותה:**

```typescript
import { anthropic } from "@workspace/integrations-anthropic-ai";

// Cache פשוט למניעת קריאות כפולות לאותה הודעה
const intentCache = new Map<string, AgentType>();

export async function classifyIntentWithAI(task: string): Promise<AgentType> {
  // בדוק cache קודם
  if (intentCache.has(task)) return intentCache.get(task)!;

  // Fallback מהיר ל-regex לביטויים ברורים
  const lower = task.toLowerCase();
  if (/error|bug|שגיאה|לא עובד|crash|fix|תקן/.test(lower)) return "debug";
  if (/deploy|netlify|vercel|פרסם|העלה/.test(lower)) return "devops";
  if (/test|בדוק|playwright|jest|qa/.test(lower)) return "qa";

  // LLM classification לכל השאר
  try {
    const res = await anthropic.messages.create({
      model: "claude-haiku-4-5-20251001",
      max_tokens: 10,
      system: `Classify the user's development request into exactly one category.
Return ONLY one word: frontend | backend | architect | debug | devops | qa
- frontend: UI, design, components, styling, animations
- backend: API, database, auth, server, queries
- architect: full app, complex system, full-stack, new project
- debug: errors, bugs, crashes, not working
- devops: deploy, hosting, CI/CD, environment
- qa: testing, review, quality check`,
      messages: [{ role: "user", content: task }],
    });

    const text =
      res.content[0].type === "text"
        ? res.content[0].text.trim().toLowerCase()
        : "";
    const validTypes: AgentType[] = [
      "frontend",
      "backend",
      "architect",
      "debug",
      "devops",
      "qa",
    ];
    const result: AgentType = validTypes.includes(text as AgentType)
      ? (text as AgentType)
      : "frontend";

    intentCache.set(task, result);
    // נקה cache אחרי 5 דקות
    setTimeout(() => intentCache.delete(task), 5 * 60 * 1000);
    return result;
  } catch {
    return "frontend"; // fallback בטוח
  }
}

// שמור את הגרסה הישנה לשימוש סינכרוני כ-fallback
export function classifyIntent(task: string): AgentType {
  const lower = task.toLowerCase();
  if (
    /error|bug|broken|crash|fix|not work|failing|exception|undefined|null|cannot read/i.test(
      lower,
    )
  )
    return "debug";
  if (
    /api|server|database|db|sql|endpoint|route|auth|login|backend|fetch|request|response/i.test(
      lower,
    )
  )
    return "backend";
  if (
    /deploy|netlify|vercel|docker|ci|cd|environment|env var|domain|hosting|publish/i.test(
      lower,
    )
  )
    return "devops";
  if (
    /test|review|check|validate|quality|qa|jest|vitest|playwright|accessibility|a11y/i.test(
      lower,
    )
  )
    return "qa";
  if (
    /architect|design|structure|plan|build.*full|create.*app|full.?stack|system|scaffold/i.test(
      lower,
    )
  )
    return "architect";
  return "frontend";
}
```

## תיקון 5 — Token-aware Context Trimming

**קובץ:** `artifacts/api-server/src/routes/projects/messages.ts`
**בעיה:** context trimming חותך לפי מספר הודעות ("last 20 messages") ולא לפי tokens בפועל. הודעה אחת עם 10K תווים שווה ל-100 הודעות קצרות.

**הוסף קובץ חדש: `services/messages/context-trimmer.ts`**

```typescript
// artifacts/api-server/src/services/messages/context-trimmer.ts

// קירוב: 1 token ≈ 4 תווים (אנגלית), 3 תווים (עברית)
function estimateTokens(text: string): number {
  const hebrewChars = (text.match(/[\u0590-\u05FF]/g) ?? []).length;
  const otherChars = text.length - hebrewChars;
  return Math.ceil(hebrewChars / 3 + otherChars / 4);
}

interface Message {
  role: string;
  content: string;
}

export function trimMessagesToTokenBudget(
  messages: Message[],
  maxTokens: number = 60_000,
): Message[] {
  let total = 0;
  const result: Message[] = [];

  // עבור מהסוף להתחלה — שמור תמיד את ההודעות האחרונות
  for (let i = messages.length - 1; i >= 0; i--) {
    const msg = messages[i]!;
    const tokens = estimateTokens(msg.content);
    if (total + tokens > maxTokens) break;
    total += tokens;
    result.unshift(msg);
  }

  return result;
}
```

**מצא בתוך messages.ts את הקוד שחותך הודעות (משהו כמו `slice(-20)` או `existingMessages.slice`) והחלף:**

```typescript

// אחרי:
```

## תיקון 7 — Alert על עלות AI קרובה למגבלה

**קובץ:** `artifacts/api-server/src/routes/projects/cost.ts` + קובץ חדש
**בעיה:** אין התראה כשמשתמש מגיע ל-95% מהמגבלה היומית. המשתמש מופתע כשהוא נחסם.

**הוסף קובץ חדש: `services/notifications/usage-alert.ts`**

```typescript
// artifacts/api-server/src/services/notifications/usage-alert.ts
import { logger } from "../../lib/logger";

const ALERT_THRESHOLD = 0.95; // 95%
// מניעת שליחת alert כפול לאותו משתמש באותו יום
const alertedToday = new Map<string, number>(); // userId → date (YYYYMMDD)

function todayStr(): string {
  return new Date().toISOString().slice(0, 10).replace(/-/g, "");
}

export function checkUsageAlert(
  userId: string,
  currentCount: number,
  maxCount: number,
  sendEvent: (event: Record<string, unknown>) => void,
): void {
  const ratio = currentCount / maxCount;
  if (ratio < ALERT_THRESHOLD) return;

  const today = todayStr();
  if (alertedToday.get(userId) === Number(today)) return; // כבר התרענו היום

  alertedToday.set(userId, Number(today));

  const remaining = maxCount - currentCount;
  logger.info(
    { userId, currentCount, maxCount, ratio },
    "Usage alert triggered",
  );

  // שלח event ל-SSE stream של המשתמש
  sendEvent({
    type: "usage_alert",
    message: `נותרו לך ${remaining} הודעות בלבד להיום. שדרג את החשבון למגבלה גבוהה יותר.`,
    currentCount,
    maxCount,
    remaining,
    thresholdPct: Math.round(ratio * 100),
  });
}
```
